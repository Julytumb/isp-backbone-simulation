#include "ns3/core-module.h"
#include "ns3/network-module.h"
#include "ns3/internet-module.h"
#include "ns3/point-to-point-module.h"
#include "ns3/applications-module.h"
#include "ns3/ipv4-global-routing-helper.h"

// NEUE HEADER
#include <string>
#include <map> 

using namespace ns3;

NS_LOG_COMPONENT_DEFINE("MapTopologyExample");

int main(int argc, char* argv[])
{
    // --- 1. Die "Landkarte" und einen Master-Container erstellen ---
    
    // Dies ist unser "Telefonbuch", das Namen auf Knoten abbildet.
    std::map<std::string, Ptr<Node>> nodeMap;
    
    // Wir behalten diesen Container, um Aktionen auf *alle* // Knoten gleichzeitig anzuwenden (z.B. Stack installieren).
    NodeContainer allNodes;

    // --- 2. Knoten erstellen und in der Map speichern ---
    NS_LOG_INFO("Creating nodes...");

    // (Hier würdest du ALLE Knotennamen aus deiner YAML-Datei eintragen)
    std::vector<std::string> nodeNames = {
        "stu-eti-a99", "ulm-n25-1", "ulm-n26-1", "hlb-hs-1", "kue-hs-1"
        // ... und so weiter für alle 50+ Knoten
    };

    for (const std::string& name : nodeNames)
    {
        // Erstelle einen neuen, leeren Knoten
        Ptr<Node> node = CreateObject<Node>();
        
        // Speichere den Knoten in unserem "Telefonbuch"
        nodeMap[name] = node;
        
        // Füge ihn zum Master-Container hinzu
        allNodes.Add(node);
        
        // Setze den Namen auch intern in ns-3 (gut für Logs)
        Names::Add(name, node);
    }
    
    // Jetzt existieren alle Knoten und sind über ihren Namen auffindbar!
    NS_LOG_INFO("Created " << allNodes.GetN() << " nodes.");

   // --- 3. Internet-Stack installieren ---
    InternetStackHelper stack;
    stack.Install(allNodes); // Installiert den Stack auf ALLEN Knoten

    // --- 4. Links definieren (beim Namen) ---
    NS_LOG_INFO("Creating links by name...");

    // *** KORREKTUR 1: Helfer-Objekte HIER deklarieren ***
    PointToPointHelper p2p;
    Ipv4AddressHelper address; // <-- DIESE ZEILE HAT GEFEHLT

    // Setze die Standard-Attribute für den p2p-Helfer
    p2p.SetDeviceAttribute("DataRate", StringValue("10Gbps")); // <-- KORREKTUR 2: Das '+' ist weg
    p2p.SetChannelAttribute("Delay", StringValue("1ms")); // Latenz hinzugefügt

    // ---
    // Link 1 (aus YAML): stu-eti-a99 <--> ulm-n25-1
    // ---
    
    // HINWEIS: Stelle sicher, dass "stu-eti-a99" in deiner nodeNames-Liste steht!
    NodeContainer link1(nodeMap["stu-eti-a99"], nodeMap["ulm-n25-1"]); 
    NetDeviceContainer devices1 = p2p.Install(link1);

    // *** KORREKTUR 3: Ein echtes Subnetz verwenden ***
    address.SetBase("10.1.1.0", "255.255.255.0"); // Subnetz 1
    Ipv4InterfaceContainer interfaces1 = address.Assign(devices1);
    // (stu-eti-a99 bekommt 10.1.1.1, ulm-n25-1 bekommt 10.1.1.2)

    // ---
    // Link 2
    // ---
    
    NodeContainer link2(nodeMap["ulm-n25-1"], nodeMap["ulm-n26-1"]); 
    NetDeviceContainer devices2 = p2p.Install(link2);

    address.SetBase("10.1.2.0", "255.255.255.0");
    Ipv4InterfaceContainer interfaces2 = address.Assign(devices2);

    // ---
    // Link 3
    // ---
    
    NodeContainer link3(nodeMap["hlb-hs-1"], nodeMap["kue-hs-1"]); 
    NetDeviceContainer devices3 = p2p.Install(link3);

    address.SetBase("10.1.3.0", "255.255.255.0");
    Ipv4InterfaceContainer interfaces3 = address.Assign(devices3);

    
    // --- 5. Routing aktivieren ---
    // (Muss NACH der IP-Zuweisung, aber VOR den Apps passieren)
    NS_LOG_INFO("Populating routing tables...");
    Ipv4GlobalRoutingHelper::PopulateRoutingTables();

    // --- 6. Test-Traffic (Applikation) hinzufügen ---
    NS_LOG_INFO("Creating applications...");
    
    // SERVER (Echo) auf Knoten "ulm-n26-1"
    // Er lauscht auf Port 9
    UdpEchoServerHelper echoServer(9); 
    
    // Installiere den Server auf dem Knoten "ulm-n26-1"
    ApplicationContainer serverApps = echoServer.Install(nodeMap["ulm-n26-1"]);
    serverApps.Start(Seconds(1.0)); // Starte Server nach 1 Sekunde
    serverApps.Stop(Seconds(10.0)); // Stoppe ihn nach 10 Sekunden

    // CLIENT auf Knoten "stu-eti-a99"
    
    // 1. Finde die IP-Adresse des Servers ("ulm-n26-1")
    //    Wir haben sie in 'interfaces2' gespeichert. Das Gerät mit Index 1 
    //    (das zweite Gerät auf link2) ist "ulm-n26-1".
    Ipv4Address serverIp = interfaces2.GetAddress(1); // Das ist die IP 10.1.2.2
    
    // 2. Erstelle den Client, der Pakete an diese IP und Port 9 sendet
    UdpEchoClientHelper echoClient(serverIp, 9);
    echoClient.SetAttribute("MaxPackets", UintegerValue(3)); // Sende 3 Pakete
    echoClient.SetAttribute("Interval", TimeValue(Seconds(1.0))); // Sende 1 Paket pro Sekunde
    echoClient.SetAttribute("PacketSize", UintegerValue(1024));

    // Installiere den Client auf dem Knoten "stu-eti-a99"
    ApplicationContainer clientApps = echoClient.Install(nodeMap["stu-eti-a99"]);
    clientApps.Start(Seconds(2.0)); // Starte 1 Sekunde nach dem Server
    clientApps.Stop(Seconds(10.0));

    // --- 7. Logging aktivieren (damit wir den Traffic sehen) ---
    LogComponentEnable("UdpEchoClientApplication", LOG_LEVEL_INFO);
    LogComponentEnable("UdpEchoServerApplication", LOG_LEVEL_INFO);

    // --- 8. Simulation starten ---
    NS_LOG_INFO("Running simulation...");
    Simulator::Run();
    Simulator::Destroy();
    NS_LOG_INFO("Done.");
}
